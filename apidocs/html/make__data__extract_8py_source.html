48 def uriParser(source):
49  """
50  Parse a URI into it's components
51 
52  Args:
53  source (str): The URI for the database connection
54 
55  Returns:
56  (dict): The parse URI
57  """
58  dbhost = None
59  dbname = None
60  dbuser = None
61  dbpass = None
62  dbport = None
63 
64  # if dbhost is 'localhost' then this tries to
65  # connect to that hostname's tcp/ip port. If dbhost
66  # is None, the datbase connection is done locally
67  # through the named pipe.
68  colon = source.find(':')
69  rcolon = source.rfind(':')
70  atsign = source.find('@')
71  slash = source.find('/')
72  # If nothing but a string, then it's a local postgres database
73  # that doesn't require a user or password to login.
74  if colon < 0 and atsign < 0 and slash < 0:
75  dbname = source
76  # Get the database name, which is always after the slash
77  if slash > 0:
78  dbname = source[slash+1:]
79  # The user field is either between the beginning of the string,
80  # and either a colon or atsign as the end.
81  if colon > 0:
82  dbuser = source[:colon]
83  if colon < 0 and atsign > 0:
84  dbuser = source[:atsign]
85  # The password field is between a colon and the atsign
86  if colon > 0 and atsign > 0:
87  dbpass = source[colon+1:atsign]
88  # The hostname for the database is after an atsign, and ends
89  # either with the end of the string or a slash.
90  if atsign > 0:
91  if rcolon > 0 and rcolon > atsign:
92  dbhost = source[atsign+1:rcolon]
93  elif slash > 0:
94  dbhost = source[atsign+1:slash]
95  else:
96  dbhost = source[atsign+1:]
97  # rcolon is only above zero if there is a port number
98  if rcolon > 0 and rcolon > atsign:
99  if slash > 0:
100  dbport = source[rcolon+1:slash]
101  else:
102  dbport = source[rcolon+1:]
103  # import epdb; epdb.st()
104  if colon > 0 and atsign < 0 and slash > 0:
105  dbpass = source[colon+1:slash]
106 
107  # print(f"{source}\n\tcolon={colon} rcolon={rcolon} atsign={atsign} slash={slash}")
108  db = {'dbname': dbname, 'dbhost': dbhost, 'dbuser': dbuser, 'dbpass': dbpass, 'dbport': dbport}
109 
110  return db

180  category: str,
181  boundary,
182  poly: bool = False,
183  ):
184  """
185  Create the JSON file used to query the Underpass database
186 
187  Args:
188  category (str): The category form the XLSForms library
189  boundary (Feature): The project AOI
190  poly: Whether to have the full geometry or just centroids returns
191 
192  Returns
193  (json): The JSON query for the Underpass database
194  """
195  path = xlsforms_path.replace("xlsforms", "data_models")
196  file = open(f"{path}/{category}.yaml", "r").read()
197  data = yaml.load(file, Loader=yaml.Loader)
198 
199  features = dict()
200  features['geometry'] = boundary
201 
202  # The database tables to query
203  # if tags exists, then only query those fields
204  columns = dict()
205  tags = data['where']['tags'][0]
206  for tag, value in tags.items():
207  if value == "not null":
208  columns[tag] = []
209  else:
210  columns[tag] = value
211  filters = {"tags": {"all_geometry": {"join_or": columns}}}
212  features['filters'] = filters
213  tables = list()
214  for table in data['from']:
215  if table == "nodes":
216  tables.append("point")
217  elif table == "ways_poly":
218  tables.append("polygon")
219  elif table == "ways_line":
220  tables.append("line")
221  elif table == "relations":
222  tables.append("line")
223  features["geometryType"] = tables
224  if not poly and category != 'highways':
225  features["centroid"] = "true"
226  return json.dumps(features)

228  def createSQL(self,
229  category: str,
230  polygon: bool = False,
231  ):
232  """
233  Create the SQL used to query a postgres database
234 
235  Args:
236  category (str): The category form the XLSForms library
237  polygon: Whether to have the full geometry or just centroids returns
238 
239  Returns
240  (list): The SQL queries for postgres using the Underpass database schema
241  """
242  path = xlsforms_path.replace("xlsforms", "data_models")
243  file = open(f"{path}/{category}.yaml", "r").read()
244  data = yaml.load(file, Loader=yaml.Loader)
245 
246  sql = list()
247  # The database tables to query
248  tables = data['from']
249  for table in tables:
250  query = "SELECT "
251  select = data['select']
252  if polygon or category == "highways":
253  centroid = "geom"
254  else:
255  centroid = "ST_Centroid(geom)"
256  # if tags exists, then only return those fields
257  if 'tags' in select:
258  for tag in select['tags']:
259  query += f" {select[tag]} AS {tag}, "
260  query += f"osm_id AS id, ST_AsEWKT({centroid}, version "
261  else:
262  query += f"osm_id AS id, ST_AsEWKT({centroid}), tags, version "
263  query += f" FROM {table} "
264  where = data['where']
265  # if tags exists, then only query those fields
266  if 'where' in data:
267  query += " WHERE "
268  tags = data['where']['tags'][0]
269  for tag, value in tags.items():
270  if value == "not null":
271  query += f"tags->>\'{tag}\' IS NOT NULL OR "
272  else:
273  # in the yaml file, multiple values for the same tag
274  # are a list
275  for val in value:
276  query += f"tags->>\'{tag}\'=\'{val}\' OR "
277  sql.append(query[:-4])
278  return sql

299  self.dbcursordbcursor.execute(sql)
300  sql = f"DROP VIEW IF EXISTS lines_view;CREATE VIEW lines_view AS SELECT * FROM ways_line WHERE ST_CONTAINS(ST_GeomFromEWKT('SRID=4326;{ewkt.wkt}'), geom)"
301  self.dbcursordbcursor.execute(sql)
302  sql = f"DROP VIEW IF EXISTS relations_view;CREATE TEMP VIEW relations_view AS SELECT * FROM nodes WHERE ST_CONTAINS(ST_GeomFromEWKT('SRID=4326;{ewkt.wkt}'), geom)"

305  if query.find(" ways_poly ") > 0:
306  query = query.replace("ways_poly", "ways_view")
307  if query.find(" ways_line ") > 0:
308  query = query.replace("ways_line", "lines_view")
309  elif query.find(" nodes ") > 0:
310  query = query.replace("nodes", "nodes_view")
311  features = list()
312  log.debug(query)
313  self.dbcursordbcursor.execute(query)
314  result = self.dbcursordbcursor.fetchall()
315  log.info("Query returned %d records" % len(result))
316  for item in result:
317  if len(item) <= 1:
318  break

321  poi = Point((float(gps[0]), float(gps[1])))

323  gps = item[1][10:]
324  poi = wkt.loads(gps)
325  tags = item[2]
326  tags["id"] = item[0]
327  tags['version'] = item[3]
328  if "name:en" in tags:
329  tags["title"] = tags["name:en"]
330  tags["label"] = tags["name:en"]
331  elif "name" in tags:
332  tags["title"] = tags["name"]
333  tags["label"] = tags["name"]
334  else:
335  tags["title"] = tags["id"]
336  tags["label"] = tags["id"]
337  features.append(Feature(geometry=poi, properties=tags))
338  return features

340  def queryRemote(self,
341  query: str = None,
342  ):
343  """
344  Query the remote the Underpass database
345 
346  Args:
347  query (str): The SQL query
348 
349  Returns:
350  (FeatureCollection): The Features returned from the query
351  """
352  url = f"{self.url}/snapshot/"
353  result = self.sessionsession.post(url, data=query, headers=self.headersheaders)
354  if result.status_code != 200:
355  log.error(f"{result.json()['detail'][0]['msg']}")
356  return None
357  task_id = result.json()['task_id']
358  newurl = f"{self.url}/tasks/status/{task_id}"
359  while True:
360  result = self.sessionsession.get(newurl, headers=self.headersheaders)
361  if result.json()['status'] == "PENDING":
362  log.debug("Retrying...")
363  time.sleep(1)
364  elif result.json()['status'] == "SUCCESS":
365  break
366  zip = result.json()['result']['download_url']
367  result = self.sessionsession.get(zip, headers=self.headersheaders)
368  fp = BytesIO(result.content)
369  zfp = zipfile.ZipFile(fp, "r")
370  zfp.extract("Export.geojson", "/tmp/")
371  # Now take that taskid and hit /tasks/status url with get
372  data = zfp.read("Export.geojson")
373  os.remove("/tmp/Export.geojson")
374  return json.loads(data)

539  poly += f"{coords[1]} {coords[0]}"

545  poly = list()
546  for coords in way.attributes['geometry']:
547  lat = coords['lat']
548  lon = coords['lon']

554 
555  collection = FeatureCollection(features)

558  file = f"{xlsforms_path}/{xlsfile}"

560  cleaned.parse(file)
561  else:
562  cleaned.parse(f"{file}x")
563  new = cleaned.cleanData(collection)
564  jsonfile = open(filespec, "w")
565  dump(new, jsonfile)
566  return new
